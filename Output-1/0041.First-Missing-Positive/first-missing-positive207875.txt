// https://leetcode.com/problems/first-missing-positive/solutions/207875/rust/
fn first_missing_positive(nums: &[i32]) -> i32 {
    let n = nums.len();
    let mut seen = vec![false; n];

    for &num in nums {
        let num = num as usize;
        if num > 0 && num <= n {
            seen[num - 1] = true;
        }
    }

    seen.into_iter()
        .position(|b| !b)
        .unwrap_or(n) as i32
        + 1
}
