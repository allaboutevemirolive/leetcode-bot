// https://leetcode.com/problems/first-missing-positive/solutions/642358/is-it-o-n-solution-in-rust/
impl Solution {
    pub fn first_missing_positive(mut nums: Vec<i32>) -> i32 {
        nums.sort();
        let mut result: i32 = 0;
        for i in 0..nums.len() {
            if nums[i] <= 0 {
                continue;
            } else if nums[i] - 1 > result {
                return result+1
            }
            result = nums[i];
        }
        return result+1;
    }
}
