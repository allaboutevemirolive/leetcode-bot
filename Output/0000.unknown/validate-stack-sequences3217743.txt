// https://leetcode.com/problems/validate-stack-sequences/solutions/3217743/rust-elixir-simulation/
impl Solution {
    pub fn validate_stack_sequences(pushed: Vec<i32>, popped: Vec<i32>) -> bool {
        let mut it = pushed.iter();
        let mut stack = Vec::new();
        for &pop_ele in popped.iter() {
            while stack.last() != Some(&pop_ele) {
                match it.next() {
                    Some(&x) => stack.push(x),
                    None => return false,
                };
            }
            stack.pop();
        }
        stack.is_empty()
    }
}
