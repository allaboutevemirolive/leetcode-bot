// https://leetcode.com/problems/validate-stack-sequences/solutions/3411887/greedy-pop/
impl Solution {
    pub fn validate_stack_sequences(pushed: Vec<i32>, popped: Vec<i32>) -> bool {
        let mut s = vec![];
        let n = pushed.len();
        let mut push = 0;
        let mut pop = 0;
        loop {
            if pop < n && s.len() > 0 && s[s.len()-1] == popped[pop] {
                pop += 1;
                s.pop();
                continue;
            }
            if push < n {
                s.push(pushed[push]);
                push +=1;
                continue;
            }
            break;
        }

        push == n && pop == n && s.len() == 0
    }
}
