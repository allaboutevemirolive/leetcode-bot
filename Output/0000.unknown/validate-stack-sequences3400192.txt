// https://leetcode.com/problems/validate-stack-sequences/solutions/3400192/rust-solution/
impl Solution {
    pub fn validate_stack_sequences(pushed: Vec<i32>, popped: Vec<i32>) -> bool {
        let mut stack = vec![];
        let mut j = 0;
        for i in 0..pushed.len() {
            stack.push(pushed[i]);
            while j < popped.len() && !stack.is_empty() && *stack.last().unwrap() == popped[j] {
                stack.pop();
                j += 1;
            }
        }
        j == popped.len()        
    }
}
