// https://leetcode.com/problems/validate-stack-sequences/solutions/3410609/rust-greedy-stack-based-solution-100-fastest/
impl Solution {
    pub fn validate_stack_sequences(pushed: Vec<i32>, popped: Vec<i32>) -> bool {
        let mut r = Vec::new();
        let mut popped_index = 0;
        for &i in pushed.iter() {
            r.push(i);
            while let Some(&x) = r.last() {
                if x == popped[popped_index] {
                    r.pop();
                    popped_index += 1;
                } else {
                    break;
                }
            }
        }
        r.len() == 0
    }
}
