// https://leetcode.com/problems/validate-stack-sequences/solutions/3410621/rust-simulate-stack/
impl Solution {
    pub fn validate_stack_sequences(pushed: Vec<i32>, popped: Vec<i32>) -> bool {
        let mut stack: Vec<i32> = Vec::new();
        let mut i = 0;

        for x in pushed {
            stack.push(x);

            while stack.last().map_or(false, |&v| v == popped[i]) {
                stack.pop();
                i += 1;
            }
        }

        i == popped.len()
    }
}
