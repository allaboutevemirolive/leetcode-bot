// https://leetcode.com/problems/validate-stack-sequences/solutions/3411075/rust/
impl Solution {
    pub fn validate_stack_sequences(pushed: Vec<i32>, popped: Vec<i32>) -> bool {
        let mut stack = vec![];
        let n = pushed.len();
        let mut i = 0;
        let mut j = 0;

        while i < n && j < n {
            while i < n && stack.last() != Some(&popped[j]) {
                stack.push(pushed[i]);
                i += 1;
            }
            while j < n && stack.last() == Some(&popped[j]) {
                stack.pop();
                j += 1;
            }
        }

        stack.is_empty()
    }
}
