// https://leetcode.com/problems/first-missing-positive/solutions/1780827/simple-rust-solution-96-and-81/
impl Solution {
    pub fn first_missing_positive(nums: Vec<i32>) -> i32 { 
        let mut missing = 1;
        let mut nums_copy = nums.clone();
        nums_copy.sort(); 
        for i in 0..nums_copy.len() {
            if ( nums_copy[i] == missing ) {
                missing += 1;
            }            
        } 
        missing
    }
}
