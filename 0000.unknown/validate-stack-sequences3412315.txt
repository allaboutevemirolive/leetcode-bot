// https://leetcode.com/problems/validate-stack-sequences/solutions/3412315/rust-simulation-explained/
impl Solution {
    pub fn validate_stack_sequences(pushed: Vec<i32>, popped: Vec<i32>) -> bool {
        let len = pushed.len();

        let mut sim = Vec::new();
        let mut j = 0;
        for i in 0..len {
            sim.push(pushed[i]);
            while !sim.is_empty() && *sim.last().unwrap() == popped[j] {
                sim.pop();
                j += 1;
            }
        }

        sim.is_empty()
    }
}
