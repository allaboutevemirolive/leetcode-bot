// https://leetcode.com/problems/validate-stack-sequences/solutions/3412024/rust-o-n/
impl Solution {
    pub fn validate_stack_sequences(pushed: Vec<i32>, popped: Vec<i32>) -> bool {
        let mut stack = vec![pushed[0]];
        let size = pushed.len();
        let mut i_pushed = 1;
        let mut i_poped = 0;

        while i_poped < size {
            if stack.is_empty() {
                stack.push(pushed[i_pushed]);
                i_pushed += 1;
            } else {
                if *stack.last().unwrap() == popped[i_poped] {
                    i_poped += 1;
                    stack.pop();
                } else {
                    if i_pushed < size {
                        stack.push(pushed[i_pushed]);
                        i_pushed += 1;
                    } else {
                        return false;
                    }
                }
            }
        }

        return true;
    }
}
