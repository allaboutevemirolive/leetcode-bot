// https://leetcode.com/problems/validate-stack-sequences/solutions/3413294/rust-stack-simulation-faster-than-100/
impl Solution {
    pub fn validate_stack_sequences(pushed: Vec<i32>, popped: Vec<i32>) -> bool {
        let len = pushed.len();
        let mut stack = Vec::with_capacity(len);
        let (mut i, mut j) = (0, 0);
        while i < len {
            if stack.is_empty() {
                stack.push(pushed[i]);
                i += 1;
            } else if stack.last().unwrap() == &popped[j] {
                stack.pop();
                j += 1;
            } else {
                stack.push(pushed[i]);
                i += 1;
            }
        }
        while j < len {
            if stack.is_empty() || stack.last().unwrap() != &popped[j] {
                return false;
            }
            stack.pop();
            j += 1;
        }
        true
    }
}
