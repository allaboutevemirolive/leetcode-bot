// https://leetcode.com/problems/validate-stack-sequences/solutions/3410639/rust-simple-stack-solution/
impl Solution {
    pub fn validate_stack_sequences(pushed: Vec<i32>, popped: Vec<i32>) -> bool {
        let mut stack:Vec<i32> = vec![];
        let mut i = 0;
        for elm in pushed {
//push elm to stack
            stack.push(elm);
            loop {
//if top of stack is equal to popped[i] we pop and increment i
                if let Some(v)=stack.last(){
                    if *v==popped[i] {
                        stack.pop();
                        i+=1;
                        continue;
                    }
                }
//otherwise we break
                break;
            }
        }
//return true if stack is empty
        stack.is_empty()
    }
}
